create table if not exists public.user_profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  email text,
  selected_portfolio text,
  quiz_answers jsonb,
  is_premium boolean default false,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

create table if not exists public.user_checklist (
  id bigint generated always as identity primary key,
  user_id uuid not null references auth.users(id) on delete cascade,
  portfolio_key text not null,
  item_id text not null,
  completed boolean not null default false,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique(user_id, portfolio_key, item_id)
);

create table if not exists public.user_pac_history (
  id bigint generated always as identity primary key,
  user_id uuid not null references auth.users(id) on delete cascade,
  portfolio_key text not null,
  month_key text not null,
  completed boolean not null default false,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique(user_id, portfolio_key, month_key)
);

create table if not exists public.user_holdings (
  id bigint generated always as identity primary key,
  user_id uuid not null references auth.users(id) on delete cascade,
  holding_key text not null,
  category text not null,
  asset_name text not null,
  isin text not null,
  amount numeric not null default 0,
  created_at timestamptz default now(),
  updated_at timestamptz default now(),
  unique(user_id, holding_key)
);
alter table public.user_profiles enable row level security;
alter table public.user_checklist enable row level security;
alter table public.user_pac_history enable row level security;
alter table public.user_holdings enable row level security;
